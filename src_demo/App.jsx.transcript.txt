53:   ha.push({ time: candles[0].time, open: prevHaOpen, high: candles[0].high, low: candles[0].low, close: prevHaClose });

84:   if (INTERVAL_SECONDS_MAP[interval]) return INTERVAL_SECONDS_MAP[interval];

238: function distanceToLineSegment(px, py, x1, y1, x2, y2) {
239:   const A = px - x1, B = py - y1, C = x2 - x1, D = y2 - y1;
240:   const dot = A * C + B * D;
241:   const len_sq = C * C + D * D;
242:   let param = -1;
243:   if (len_sq !== 0) param = dot / len_sq;
244:   let xx, yy;
245:   if (param < 0) { xx = x1; yy = y1; }
246:   else if (param > 1) { xx = x2; yy = y2; }
247:   else { xx = x1 + param * C; yy = y1 + param * D; }
248:   const dx = px - xx, dy = py - yy;
249:   return Math.sqrt(dx * dx + dy * dy);
250: }
